const ContrastColor = compute({
id: 'contrast-color',
get: ({ get }) => {
const color = get(ColorAtom);
const rgba = color.replace(/[rgba\(\)]/g, '').split(',');

...rgba.slice(0, 3).map((item) => 255 - Number(item)),

const HighLightColor = compute({
id: 'highlight-color',
get: ({ get }) => {
const color = get(ColorAtom);
const rgba = color.replace(/[rgba?\(\)]/g, '').split(',');

},
});
