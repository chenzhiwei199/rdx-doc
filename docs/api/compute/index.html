import React from 'react';
import { atom, useRdxState, compute, RdxContext } from '@alife/rdx';
const unit = atom({
id: 'unit',
defaultValue: 10,
});
const amount = atom({
id: 'amount',
defaultValue: 20,
});
const total = compute<number>({
id: 'total',
get: ({ get }) => {
const total = get(unit) * get(amount);
return total;
},
set: ({ get, set }, newValue) => {
set(amount, newValue / get(unit));
},
});
export const Root = () => {
return (
